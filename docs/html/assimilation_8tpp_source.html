119  //double GPP_plant = 0, Rl_plant = 0, dpsi_avg = 0;
120  double fapar = 1-exp(-par.k_light*G->lai);
121  bool by_layer = false;
122 
123  plant_assim.gpp = 0;
124  plant_assim.rleaf = 0;
125  plant_assim.trans = 0;
126  plant_assim.dpsi_avg = 0;
127  plant_assim.vcmax_avg = 0;
128  plant_assim.vcmax25_avg = 0;
129  plant_assim.mc_avg = 0;
130  plant_assim.gs_avg = 0;
131  plant_assim.c_open_avg = 0;
132 
133  double ca_cumm = 0;
134  //std::cout << "--- PPA Assim begin ---" << "\n";
135  for (int ilayer=0; ilayer <= env.n_layers; ++ilayer){ // for l in 1:layers{
136  double zst = env.z_star[ilayer];
137  double ca_layer = G->crown_area_above(zst, traits) - ca_cumm;
138  //std::cout << "h = " << G->height << ", z* = " << zst << ", I = " << env.canopy_openness[ilayer] << ", fapar = " << fapar << /*", A = " << (res.a + res.vcmax*par.rd) << " umol/m2/s x " <<*/ ", ca_layer = " << ca_layer << /*" m2 = " << (res.a + res.vcmax*par.rd) * ca_layer << ", vcmax = " << res.vcmax <<*/ "\n";
139 
140  if (by_layer == true){
141  auto res = leaf_assimilation_rate(env.canopy_openness[ilayer], fapar, env, par, traits);
142  plant_assim.gpp += (res.a + res.vcmax*par.rd) * ca_layer;
143  plant_assim.rleaf += (res.vcmax*par.rd) * ca_layer;
144  plant_assim.trans += res.e * ca_layer;
145  plant_assim.dpsi_avg += res.dpsi * ca_layer;
146  plant_assim.vcmax_avg += res.vcmax * ca_layer;
147  plant_assim.gs_avg += res.gs * ca_layer;
148  plant_assim.vcmax25_avg += res.vcmax25 * ca_layer;
149  plant_assim.mc_avg += res.mc * ca_layer;
150  }
151 
152  plant_assim.c_open_avg += env.canopy_openness[ilayer] * ca_layer;
153  ca_cumm += ca_layer;
154 
155  }
156  assert(fabs(ca_cumm/G->crown_area - 1) < 1e-6);
157  double ca_total = G->crown_area; // total crown area
158  plant_assim.c_open_avg /= ca_total; // unitless
159  if (by_layer == true){
160  plant_assim.dpsi_avg /= ca_total; // MPa
161  plant_assim.vcmax_avg /= ca_total; // umol CO2/m2/s
162  plant_assim.gs_avg /= ca_total; // mol CO2/m2/s
163  plant_assim.vcmax25_avg /= ca_total; // umol CO2/m2/s
164  plant_assim.mc_avg /= ca_total; // unitless
165  //std::cout << "--- total (by layer) \n";
166  //std::cout << "h = " << G->height << ", nz* = " << env.n_layers << ", I = " << plant_assim.c_open_avg << ", fapar = " << fapar << ", A = " << plant_assim.gpp/ca_total << " umol/m2/s x " << ca_total << " = " << plant_assim.gpp << ", vcmax_avg = " << plant_assim.vcmax_avg << "\n";
167  }
168 
169  if (by_layer == false){
170  auto res = leaf_assimilation_rate(plant_assim.c_open_avg, fapar, env, par, traits);
171  plant_assim.gpp = (res.a + res.vcmax*par.rd) * ca_total;
172  plant_assim.rleaf = (res.vcmax*par.rd) * ca_total;
173  plant_assim.trans = res.e * ca_total;
174  plant_assim.dpsi_avg = res.dpsi;
175  plant_assim.vcmax_avg = res.vcmax;
176  plant_assim.gs_avg = res.gs;
177  plant_assim.vcmax25_avg = res.vcmax25;
178  plant_assim.mc_avg = res.mc;

182  }
183  //std::cout << "---\nCA traversed = " << ca_cumm << " -- " << G->crown_area << "\n";
184 
185  // Convert units from per sec to per unit_t (unit_t is the unit in which time is counted, e.g. yr, day)
186  double sec_per_unit_t = 86400*par.days_per_tunit; // s-1 ---> unit_t-1
187 
188  plant_assim.gpp *= (sec_per_unit_t * 1e-6 * par.cbio); // umol co2/s ----> umol co2/unit_t --> mol co2/unit_t --> kg/unit_t

209 
210  double A = plant_assim.gpp;
211  double R = plant_assim.rleaf + plant_assim.rroot + plant_assim.rstem;
212  double T = plant_assim.tleaf + plant_assim.troot;
213 
214  plant_assim.npp = par.y*(A-R) - T; // net biomass growth rate (kg unit_t-1)
215  if (std::isnan(plant_assim.npp)){
216  std::cout << "ART = " << A << " " << R << " " << T << '\n';
217  throw std::runtime_error("assimilator npp is nan");
218  }
219
