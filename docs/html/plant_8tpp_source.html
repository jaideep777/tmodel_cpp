17  double dE_dL = (res_plus.trans - res.trans)/geometry.crown_area/par.dl;
18 // double ddpsi_dL = dE_dL * viscosity / (traits.K_xylem * phydro::P(env.clim_inst.swp, traits.p50_xylem, traits.b_xylem)); // FIXME: Need proper unit conversion
19 
20  double dL_dt = 0;
21  if (par.optimize_lai) dL_dt = par.response_intensity*(dnpp_dL - par.Chyd*dE_dL - par.Cc*traits.lma); // FIXME: This condition can be applied to the whole block
22  //std::cout << "dnpp_dL = " << dnpp_dL << ", dE_dL = " << 0.001*dE_dL << ", Cc = " << traits.K_leaf << "\n";
23 
24  if (lai_curr < 0.1) dL_dt = 0; // limit to prevent LAI going negative
25 
26  // calculate and constrain rate of LAI change
27  double max_alloc_lai = par.max_alloc_lai * _dmass_dt_tot; // if npp is negative, there can be no lai increment. if npp is positive, max 10% can be allocated to lai increment
28  bp.dmass_dt_lai = geometry.dmass_dt_lai(dL_dt, max_alloc_lai, traits); // biomass change resulting from LAI change

194 template<class Env>
195 void Plant::partition_biomass(double dm_dt_tot, double dm_dt_lai, Env &env){
196 
197  // NOTE: LAI increment is prioritized (already subtracted from npp above) // if lai is increasing, biomass is partitioned into lai growth and remaining components
198  // if lai is decreasing, lost biomass goes into litter
199  double dmass_dt_nonlai = dm_dt_tot - std::max(dm_dt_lai, 0.0);
200  bp.dmass_dt_lit = std::max(-dm_dt_lai, 0.0);
201 
202  // fraction of biomass going into reproduction and biomass allocation to reproduction
203  double fR = geometry.dreproduction_dmass(par, traits);
204  bp.dmass_dt_rep = fR * dmass_dt_nonlai;
205 
206  // fraction of biomass going into growth and size growth rate
207  double dmass_growth_dmass = (1-fR);
208  bp.dmass_dt_growth = dmass_growth_dmass * dmass_dt_nonlai;
209 
210  // consistency check - see that all biomass allocations add up as expected
211  double dmass_dt_allocated = bp.dmass_dt_lai + bp.dmass_dt_lit + bp.dmass_dt_rep + bp.dmass_dt_growth;
212  assert(fabs(bp.dmass_dt_tot - dmass_dt_allocated) < 1e-6);
