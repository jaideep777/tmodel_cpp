32 inline double incbeta(double a, double b, double x, double STOP=1.0e-8) {
33  const double TINY = 1.0e-30;
34 
35  if (x < 0.0 || x > 1.0) return 1.0/0.0;
36 
37  /*The continued fraction converges nicely for x < (a+1)/(a+b+2)*/
38  if (x > (a+1.0)/(a+b+2.0)) {
39  return (1.0-incbeta(b,a,1.0-x, STOP)); /*Use the fact that beta is symmetrical.*/
40  }
41 
42  /*Find the first part before the continued fraction.*/
43  const double lbeta_ab = lgamma(a)+lgamma(b)-lgamma(a+b);
44  const double front = exp(log(x)*a+log(1.0-x)*b-lbeta_ab) / a;
45 
46  /*Use Lentz's algorithm to evaluate the continued fraction.*/
47  double f = 1.0, c = 1.0, d = 0.0;
48 
49  int i, m;
50  for (i = 0; i <= 200; ++i) {
51  m = i/2;
52 
53  double numerator;
54  if (i == 0) {
55  numerator = 1.0; /*First numerator is 1.0.*/
56  } else if (i % 2 == 0) {
57  numerator = (m*(b-m)*x)/((a+2.0*m-1.0)*(a+2.0*m)); /*Even term.*/
58  } else {
59  numerator = -((a+m)*(a+b+m)*x)/((a+2.0*m)*(a+2.0*m+1)); /*Odd term.*/
60  }
61 
62  /*Do an iteration of Lentz's algorithm.*/
63  d = 1.0 + numerator * d;
64  if (fabs(d) < TINY) d = TINY;
65  d = 1.0 / d;
66 
67  c = 1.0 + numerator / c;
68  if (fabs(c) < TINY) c = TINY;
69 
70  const double cd = c*d;
71  f *= cd;
72 
73  /*Check for stop.*/
74  if (fabs(1.0-cd) < STOP) {
75  return front * (f-1.0);
76  }
77  }
78 
79  return 1.0/0.0; /*Needed more loops, did not converge.*/
80 }
