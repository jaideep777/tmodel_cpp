88 double PlantArchitecture::dsize_dmass(PlantTraits &traits) const {
89  double dh_dd = geom.a * exp(-geom.a*diameter/traits.hmat);
90  double dmleaf_dd = traits.lma * lai * geom.pic_4a * (height + diameter*dh_dd); // LAI variation is accounted for in biomass production rate
91  double dmtrunk_dd = (geom.eta_c * M_PI * traits.wood_density / 4) * (2*height + diameter*dh_dd)*diameter;
92  double dmbranches_dd = (sqrt(geom.c / geom.a) * M_PI * traits.wood_density / 12) * (2.5*height + 0.5*diameter*dh_dd) * diameter*sqrt(diameter/height);
93  double dmroot_dd = (traits.zeta/traits.lma) * dmleaf_dd;
94  double dmcroot_dd = (dmbranches_dd + dmtrunk_dd)*traits.fcr;
95 
96  double dmass_dd = dmleaf_dd + dmtrunk_dd + dmbranches_dd + dmroot_dd + dmcroot_dd;
97  return 1/dmass_dd;
98 }
